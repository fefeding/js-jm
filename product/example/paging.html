<div id="data"></div>

<div id="paging"></div>

<script type="text/javascript">

    $('#paging').paging({        

        showCount: 6, //当前页附近显示多少页码，比如，4或5表示前后显二个页码，6表示前后显示三个,都会除以2

        change: function (p) {

            showPageData(p);

        }

    });

    /**

    * 显示当前页数据

    **/

    function showPageData(index) {

        $('#data').html('第' + index + '页');

        //当点击而码后触发的事件，传入点击的是第几页

        $('#paging').paging({ index: index, count: 100 });

    }

    $(function () {

        showPageData(1);//默认显示第一页

    });

</script>
