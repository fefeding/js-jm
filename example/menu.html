点击此区域弹出菜单

 

<div id="testmenu" style="height: 100px; border: 1px solid black;">

        点击此区域弹出菜单

    </div>

    <script type="text/javascript">

        $(function () {

        //绑定菜单

            $('#testmenu').menu({

        //子菜单

         items: [

        {

            text: 'testa',//子菜单名

            click: function () {//点击事件

                alert('a');

            }

        },

        {

            text: 'testb',

            click: function () {

                alert('b');

            }

        }

        ],

                button: 2//0=左健菜单,2=右健菜单,

            });

        });

    </script>
